B站批量取消追番追剧

  • 时间: 2024-10-01

浏览器进入追番或追剧页面，开发者模式下（F12）在“控制台”使用


批量取消追番：

(function cl() {
    let e = document.querySelector("#page-bangumi > div > div.section.bangumi > div > div > ul > li:nth-child(1) > div > ul > li:nth-child(4)");

    if (e) {
        e.click();

        setTimeout(cl, 0);
    }
})();
